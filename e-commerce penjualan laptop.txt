    <script>
        // Data Keranjang Belanja (Menggunakan Local Storage)
        let cart = JSON.parse(localStorage.getItem('gamingCart')) || [];

        const cartCountElement = document.getElementById('cart-count');
        const cartItemsElement = document.getElementById('cart-items');
        const cartTotalElement = document.getElementById('cart-total');
        const addToCartButtons = document.querySelectorAll('.add-to-cart'); 

        // Elemen Modal
        const modal = document.getElementById("cart-modal");
        const cartIcon = document.querySelector(".cart-icon");
        const closeButton = document.querySelector(".close-button");

        // Fungsi untuk menyimpan keranjang ke Local Storage
        function saveCart() {
            localStorage.setItem('gamingCart', JSON.stringify(cart));
        }

        // Fungsi untuk memperbarui tampilan keranjang
        function updateCartDisplay() {
            cartItemsElement.innerHTML = ''; 
            let total = 0;

            cart.forEach(item => {
                const listItem = document.createElement('li');
                const itemPrice = item.price * item.quantity;
                
                // Format harga menjadi format Rupiah
                const formattedPrice = new Intl.NumberFormat('id-ID', {
                    style: 'currency',
                    currency: 'IDR',
                    minimumFractionDigits: 0
                }).format(itemPrice);

                listItem.innerHTML = `
                    <span>${item.name} (x${item.quantity})</span>
                    <span>${formattedPrice}</span>
                `;
                cartItemsElement.appendChild(listItem);
                total += itemPrice;
            });

            // Perbarui total dan jumlah item
            const formattedTotal = new Intl.NumberFormat('id-ID', {
                style: 'currency',
                currency: 'IDR',
                minimumFractionDigits: 0
            }).format(total);
            
            cartTotalElement.textContent = formattedTotal;
            cartCountElement.textContent = cart.reduce((sum, item) => sum + item.quantity, 0);

            saveCart();
        }

        // Fungsi untuk menambah item ke keranjang
        function addItemToCart(event) {
            const card = event.target.closest('.product-card');
            const id = card.dataset.id;
            const name = card.dataset.name;
            const price = parseInt(card.dataset.price);

            const existingItem = cart.find(item => item.id === id);

            if (existingItem) {
                existingItem.quantity += 1;
            } else {
                cart.push({ id, name, price, quantity: 1 });
            }

            alert(`${name} telah ditambahkan ke keranjang!`);
            updateCartDisplay();
        }

        // Tambahkan event listener ke semua tombol "Tambah ke Keranjang"
        addToCartButtons.forEach(button => {
            button.addEventListener('click', addItemToCart);
        });


        // LOGIKA MODAL (POP-UP KERANJANG)

        // Buka modal saat ikon keranjang diklik
        cartIcon.addEventListener('click', () => {
            modal.style.display = "block";
        });

        // Tutup modal saat tombol 'x' diklik
        closeButton.addEventListener('click', () => {
            modal.style.display = "none";
        });

        // Tutup modal saat user mengklik di luar area modal
        window.addEventListener('click', (event) => {
            if (event.target === modal) {
                modal.style.display = "none";
            }
        });
